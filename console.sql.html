pragma foreign_keys = ON;
pragma foreign_keys;
create table Temperature (
    Id integer not null unique primary key autoincrement,
    Temperature text not null
);
create table Size (
    Id integer not null unique primary key autoincrement,
    Size text not null
);
create table Address (
    Id integer not null unique primary key autoincrement,
    Country text not null,
    State text not null,
    City text not null,
    Zipcode text not null,
    Street text not null
);
create table Company (
    Id integer not null unique primary key autoincrement,
    Name text not null
);
create table SalesRep (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    Phone text not null,
    Email text
);
create table BlendType (
    Id integer not null unique primary key autoincrement,
    Type text not null
);
create table Store (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    Phone text not null,
    AddressId integer not null,
    foreign key (AddressId) references Address (Id)
);
create table Manufacturer (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    AddressId integer not null,
    CompanyId integer not null,
    SalesRepId integer not null,
    foreign key (AddressId) references Address (Id),
    foreign key (SalesRepId) references SalesRep (Id),
    foreign key (CompanyId) references Company (Id)
);
create table Cup (
    Id integer not null unique primary key autoincrement,
    DateReceived text not null,
    Quantity text not null,
    ManufacturerId integer not null,
    StoreId integer not null,
    SizeId integer not null,
    foreign key (SizeId) references Size (Id),
    foreign key (StoreId) references Store (Id),
    foreign key (ManufacturerId) references Manufacturer (Id)
);
create table Drink (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    CupId integer not null,
    TemperatureId integer not null,
    foreign key (CupId) references Cup (Id),
    foreign key (TemperatureId) references Temperature (Id)
);
create table Flavor (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    Type text not null,
    DateReceived text not null,
    ExpirationDate text not null,
    Quantity integer not null,
    ManufacturerId integer not null,
    StoreId integer not null,
    foreign key (ManufacturerId) references Manufacturer (Id),
    foreign key (StoreId) references Store (Id)
);
create table DrinkFlavor (
    Id integer,
    DrinkId integer,
    FlavorId integer,
    primary key (Id, DrinkId, FlavorId),
    foreign key (DrinkId) references Drink (Id),
    foreign key (FlavorId) references Flavor (Id)
);
create table MixIn (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    Type text not null,
    DateReceived text not null,
    ExpirationDate text not null,
    Quantity integer not null,
    ManufacturerId integer not null,
    StoreId integer not null,
    foreign key (ManufacturerId) references Manufacturer (Id),
    foreign key (StoreId) references Store (Id)
);
create table DrinkMixIn (
    Id integer,
    DrinkId integer,
    MixInId integer,
    primary key (Id, DrinkId, MixInId),
    foreign key (DrinkId) references Drink (Id),
    foreign key (MixInId) references MixIn (Id)
);
create table Milk (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    Type text not null,
    Dairy integer not null,
    DateReceived text not null,
    ExpirationDate text not null,
    Quantity integer not null,
    ManufacturerId integer not null,
    StoreId integer not null,
    foreign key (ManufacturerId) references Manufacturer (Id),
    foreign key (StoreId) references Store (Id)
);
create table DrinkMilk (
    Id integer,
    DrinkId integer,
    MilkId integer,
    primary key (Id, DrinkId, MilkId),
    foreign key (DrinkId) references Drink (Id),
    foreign key (MilkId) references Milk (Id)
);
create table Blend (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    Type text not null,
    DateReceived text not null,
    ExpirationDate text not null,
    Quantity integer not null,
    ManufacturerId integer not null,
    StoreId integer not null,
    BlendTypeId int not null,
    foreign key (ManufacturerId) references Manufacturer (Id),
    foreign key (StoreId) references Store (Id),
    foreign key (BlendTypeId) references BlendType (Id)
);
create table DrinkBlend (
    Id integer,
    DrinkId integer,
    BlendId integer,
    primary key (Id, DrinkId, BlendId),
    foreign key (DrinkId) references Drink (Id),
    foreign key (BlendId) references Blend (Id)
);
create table Topping (
    Id integer not null unique primary key autoincrement,
    Name text not null,
    Type text not null,
    DateReceived text not null,
    ExpirationDate text not null,
    Quantity integer not null,
    ManufacturerId integer not null,
    StoreId integer not null,
    foreign key (ManufacturerId) references Manufacturer (Id),
    foreign key (StoreId) references Store (Id)
);
create table DrinkTopping (
    Id integer,
    DrinkId integer,
    ToppingId integer,
    primary key (Id, DrinkId, ToppingId),
    foreign key (DrinkId) references Drink (Id),
    foreign key (ToppingId) references Topping (Id)
);
